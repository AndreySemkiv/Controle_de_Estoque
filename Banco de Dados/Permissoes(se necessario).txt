-- Cria usuário para acesso via localhost
CREATE USER IF NOT EXISTS 'estoque_user'@'localhost' IDENTIFIED BY '12345';

-- Cria usuário para acesso via 127.0.0.1
CREATE USER IF NOT EXISTS 'estoque_user'@'127.0.0.1' IDENTIFIED BY '12345';

-- Dá permissão total no banco estoque_rolos para os dois
GRANT ALL PRIVILEGES ON estoque_rolos.* TO 'estoque_user'@'localhost';
GRANT ALL PRIVILEGES ON estoque_rolos.* TO 'estoque_user'@'127.0.0.1';

-- Atualiza permissões
FLUSH PRIVILEGES;

GRANT ALL PRIVILEGES ON estoque_rolos.* TO 'estoque_user'@'127.0.0.1' IDENTIFIED BY '12345';
FLUSH PRIVILEGES;
